SELECT L.NAME, I.SUMMARY
FROM ISSUES I
JOIN ISSUESLISTS L ON I.ISSUESLIST_ID = L.ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS Tasks
FROM ISSUES I
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID
HAVING COUNT(*) > 0;
